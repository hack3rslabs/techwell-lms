const express = require('express');
const multer = require('multer');
const fs = require('fs');
const path = require('path');
const pdf = require('pdf-parse');
const { PrismaClient } = require('@prisma/client');
const { GoogleGenerativeAI } = require('@google/generative-ai');
const { authenticate, authorize } = require('../middleware/auth');

const router = express.Router();
const prisma = new PrismaClient({ datasources: { db: { url: process.env.DATABASE_URL } } });

const upload = multer({ dest: 'uploads/temp/' });

const GEMINI_KEY = process.env.GEMINI_API_KEY;
const genAI = new GoogleGenerativeAI(GEMINI_KEY || 'MISSING_KEY');

/**
 * @route   GET /api/candidates
 * @desc    Get candidates directory with search and filters
 * @access  Private (Admin/Staff/Recruiter)
 */
router.get('/', authenticate, authorize('ADMIN', 'SUPER_ADMIN', 'STAFF', 'EMPLOYER', 'RECRUITER'), async (req, res, next) => {
    try {
        const { 
            search, category, status, page = 1, limit = 10,
            experienceLevel, skills, pinCode, state, district,
            education, passedOutYear, currentCTC, expectedCTC, industry
        } = req.query;
        const pageNum = parseInt(page);
        const limitNum = parseInt(limit);

        const where = {};

        if (category) where.category = category;
        if (status) where.status = status;
        if (experienceLevel) where.experienceLevel = experienceLevel;
        if (pinCode) where.pinCode = pinCode;
        if (state) where.state = { contains: state, mode: 'insensitive' };
        if (district) where.district = { contains: district, mode: 'insensitive' };
        if (education) where.education = { contains: education, mode: 'insensitive' };
        if (passedOutYear) where.passedOutYear = passedOutYear;
        if (currentCTC) where.currentCTC = { contains: currentCTC, mode: 'insensitive' };
        if (expectedCTC) where.expectedCTC = { contains: expectedCTC, mode: 'insensitive' };
        if (industry) where.industry = { contains: industry, mode: 'insensitive' };

        if (skills) {
            // skills comes as comma-separated string: "React,Node"
            const skillsArray = skills.split(',').map(s => s.trim());
            where.skills = { hasSome: skillsArray };
        }

        if (search) {
            where.OR = [
                { name: { contains: search, mode: 'insensitive' } },
                { email: { contains: search, mode: 'insensitive' } },
                { phone: { contains: search, mode: 'insensitive' } },
                { parsedText: { contains: search, mode: 'insensitive' } }
            ];
        }

        const [candidates, total] = await Promise.all([
            prisma.candidateProfile.findMany({
                where,
                skip: (pageNum - 1) * limitNum,
                take: limitNum,
                orderBy: { createdAt: 'desc' },
                include: {
                    user: {
                        select: {
                            id: true,
                            name: true,
                            email: true,
                            role: true,
                            interviews: {
                                select: {
                                    id: true,
                                    status: true,
                                    evaluation: {
                                        select: {
                                            overallScore: true
                                        }
                                    }
                                }
                            }
                        }
                    },
                    lead: {
                        select: {
                            id: true,
                            source: true,
                            lifecycleStage: true
                        }
                    }
                }
            }),
            prisma.candidateProfile.count({ where })
        ]);

        return res.status(200).json({
            candidates,
            pagination: {
                page: pageNum,
                limit: limitNum,
                total,
                pages: Math.ceil(total / limitNum)
            }
        });
    } catch (error) {
        next(error);
    }
});

/**
 * @route   GET /api/candidates/:id
 * @desc    Get candidate profile by ID
 * @access  Private (Admin/Staff/Recruiter)
 */
router.get('/:id', authenticate, authorize('ADMIN', 'SUPER_ADMIN', 'STAFF', 'EMPLOYER', 'RECRUITER'), async (req, res, next) => {
    try {
        const candidate = await prisma.candidateProfile.findUnique({
            where: { id: req.params.id },
            include: {
                user: {
                    select: {
                        id: true,
                        name: true,
                        email: true,
                        phone: true,
                        dob: true,
                        qualification: true,
                        college: true
                    }
                },
                lead: true
            }
        });

        if (!candidate) {
            return res.status(404).json({ error: 'Candidate profile not found' });
        }

        return res.status(200).json({ candidate });
    } catch (error) {
        next(error);
    }
});

/**
 * @route   PATCH /api/candidates/:id/status
 * @desc    Update candidate progress/screening status
 * @access  Private (Admin/Staff/Recruiter)
 */
router.patch('/:id/status', authenticate, authorize('ADMIN', 'SUPER_ADMIN', 'STAFF', 'EMPLOYER', 'RECRUITER'), async (req, res, next) => {
    try {
        const { status } = req.body;
        if (!status) {
            return res.status(400).json({ error: 'Status is required' });
        }

        const updated = await prisma.candidateProfile.update({
            where: { id: req.params.id },
            data: { status }
        });

        return res.status(200).json({ message: 'Candidate status updated successfully', candidate: updated });
    } catch (error) {
        next(error);
    }
});

/**
 * @route   POST /api/candidates/onboard
 * @desc    Onboard a new candidate manually (via Onboarding Flow)
 * @access  Private (Authenticated User)
 */
router.post('/onboard', authenticate, async (req, res, next) => {
    try {
        const {
            interestType,
            experienceLevel,
            skills,
            currentCTC,
            expectedCTC,
            noticePeriod,
            interestedRole,
            category,
            pinCode,
            state,
            district,
            education,
            passedOutYear,
            industry
        } = req.body;

        const userId = req.user.id;

        // Ensure user exists
        const user = await prisma.user.findUnique({
            where: { id: userId },
            select: { name: true, email: true, phone: true }
        });

        if (!user) {
            return res.status(404).json({ error: 'User not found' });
        }

        const candidateProfile = await prisma.candidateProfile.upsert({
            where: { userId },
            update: {
                interestType,
                experienceLevel,
                skills,
                currentCTC,
                expectedCTC,
                noticePeriod,
                interestedRole,
                category: category || 'NON_IT_OTHERS',
                pinCode,
                state,
                district,
                education,
                passedOutYear,
                industry,
                status: 'SCREENING',
            },
            create: {
                userId,
                name: user.name,
                email: user.email,
                phone: user.phone || null,
                interestType,
                experienceLevel,
                skills: skills || [],
                currentCTC,
                expectedCTC,
                noticePeriod,
                interestedRole,
                category: category || 'NON_IT_OTHERS',
                pinCode,
                state,
                district,
                education,
                passedOutYear,
                industry,
                status: 'SCREENING',
            }
        });

        return res.status(201).json({
            message: 'Candidate profile created/updated successfully',
            candidate: candidateProfile
        });
    } catch (error) {
        next(error);
    }
});

/**
 * @route   POST /api/candidates/parse-upload
 * @desc    Upload & Parse resume using Gemini, auto-link to Users or CRM Leads
 * @access  Private (Admin/Staff)
 */
router.post('/parse-upload', authenticate, authorize('ADMIN', 'SUPER_ADMIN', 'STAFF'), upload.single('resume'), async (req, res, next) => {
    let tempPath = null;
    try {
        if (!req.file) {
            return res.status(400).json({ error: 'No resume file uploaded' });
        }
        tempPath = req.file.path;

        // Parse PDF to text
        let extractedText = '';
        try {
            const dataBuffer = fs.readFileSync(tempPath);
            const parsedPdf = await pdf(dataBuffer);
            extractedText = parsedPdf.text || '';
        } catch (pdfError) {
            console.error('[Resume Parser] PDF parsing failed:', pdfError);
            return res.status(400).json({ error: 'Failed to extract text from uploaded PDF resume' });
        }

        if (!extractedText.trim()) {
            return res.status(400).json({ error: 'Resume PDF contains no readable text content' });
        }

        // Call Gemini to structure and classify resume content
        const model = genAI.getGenerativeModel({ model: 'gemini-1.5-flash' });
        const prompt = `
You are an expert recruitment ATS parser and data scientist.
Your job is to read the raw text of a resume and extract the candidate profile information.

Raw Resume Text:
${extractedText.substring(0, 8000)}

Instructions:
1. Extract candidate name, email, phone (mobile number), list of skills, and education/certifications.
2. Group the candidate into one of these strict Categories based on their profile:
   - "IT_CLOUD" (DevOps, AWS, Azure, Cloud Architect, GCP)
   - "IT_DEVOPS" (Docker, Kubernetes, CI/CD, Terraform)
   - "IT_CYBER_SECURITY" (Infosec, Pentesting, SOC, Network Security)
   - "IT_NETWORKING" (Cisco, CCNA, routing, switching)
   - "IT_JAVA" (Java, Spring Boot, J2EE)
   - "IT_PYTHON" (Python, Django, Flask)
   - "IT_DATA_SCIENCE" (AI, Machine Learning, Data Analytics, Pandas)
   - "IT_FULL_STACK" (Node, React, Next, Angular, PHP, full stack web)
   - "NON_IT_HR" (Human Resources, recruiting, talent acquisition)
   - "NON_IT_FINANCE" (Accounting, financial analyst, tax)
   - "NON_IT_SALES" (Business development, sales executive, account manager)
   - "NON_IT_MARKETING" (Digital marketing, SEO, social media)
   - "NON_IT_BPO" (Customer support, call center, customer success)
   - "NON_IT_OTHERS" (For anything else)
3. Rate their ATS compatibility score (atsScore) between 10 to 100 based on standard resume formatting and layout indicators.
4. Rate their initial Readiness Score (readinessScore) between 10 to 100 based on experience and tech depth.

Return ONLY a valid JSON object matching this structure (no markdown wrappers, no \`\`\`json):
{
  "name": "Candidate Full Name",
  "email": "candidate@example.com or null",
  "phone": "candidate phone number or null",
  "skills": ["skill1", "skill2", "etc"],
  "category": "One of the categories listed above",
  "atsScore": number,
  "readinessScore": number
}
`;

        const result = await model.generateContent(prompt);
        const responseText = result.response.text();
        let cleanJson = responseText.replace(/```json/gi, '').replace(/```/g, '').trim();
        
        let parsedData;
        try {
            parsedData = JSON.parse(cleanJson);
        } catch (jsonErr) {
            console.error('[Resume Parser] Failed to parse Gemini response as JSON. Raw:', responseText);
            throw new Error('Gemini response could not be parsed as valid JSON candidate schema');
        }

        // Search for matching User or Lead to auto-link
        let userId = null;
        let leadId = null;

        if (parsedData.email) {
            const matchedUser = await prisma.user.findUnique({
                where: { email: parsedData.email },
                select: { id: true, name: true }
            });
            if (matchedUser) {
                userId = matchedUser.id;
            }

            const matchedLead = await prisma.lead.findFirst({
                where: { email: parsedData.email },
                select: { id: true }
            });
            if (matchedLead) {
                leadId = matchedLead.id;
            }
        }

        if (!leadId && parsedData.phone) {
            const matchedLeadByPhone = await prisma.lead.findFirst({
                where: { phone: { contains: parsedData.phone } },
                select: { id: true }
            });
            if (matchedLeadByPhone) {
                leadId = matchedLeadByPhone.id;
            }
        }

        // Upsert Candidate Profile
        const candidateProfile = await prisma.candidateProfile.upsert({
            where: parsedData.email ? { email: parsedData.email } : { id: 'non-existent-trigger-cuid' },
            update: {
                name: parsedData.name || 'Unknown Candidate',
                phone: parsedData.phone || null,
                skills: parsedData.skills || [],
                category: parsedData.category || 'NON_IT_OTHERS',
                parsedText: extractedText.substring(0, 5000),
                atsScore: parseFloat(parsedData.atsScore) || 50,
                readinessScore: parseFloat(parsedData.readinessScore) || 50,
                userId,
                leadId
            },
            create: {
                name: parsedData.name || 'Unknown Candidate',
                email: parsedData.email || null,
                phone: parsedData.phone || null,
                skills: parsedData.skills || [],
                category: parsedData.category || 'NON_IT_OTHERS',
                parsedText: extractedText.substring(0, 5000),
                atsScore: parseFloat(parsedData.atsScore) || 50,
                readinessScore: parseFloat(parsedData.readinessScore) || 50,
                status: 'SCREENING',
                userId,
                leadId
            }
        });

        // Update corresponding Lead lifecycle stage if linked
        if (leadId) {
            await prisma.lead.update({
                where: { id: leadId },
                data: {
                    lifecycleStage: 'PLACEMENT_POOL',
                    placementStatus: 'SCREENING'
                }
            });
        }

        return res.status(201).json({
            message: 'Resume parsed and candidate profile updated successfully',
            candidate: candidateProfile
        });
    } catch (error) {
        next(error);
    } finally {
        // Clean up temp file
        if (tempPath && fs.existsSync(tempPath)) {
            try {
                fs.unlinkSync(tempPath);
            } catch (err) {
                console.error('[Resume Parser] Failed to delete temp file:', err);
            }
        }
    }
});

/**
 * @route   GET /api/candidates/me/skill-gaps
 * @desc    AI Skill Gap & Micro-Learning Recommender
 * @access  Private (Student)
 */
router.get('/me/skill-gaps', authenticate, authorize('STUDENT'), async (req, res, next) => {
    try {
        const userId = req.user.id;

        const candidate = await prisma.candidateProfile.findUnique({
            where: { userId }
        });

        const interviews = await prisma.interview.findMany({
            where: { userId },
            include: { evaluation: true }
        });

        // If no candidate profile, can't analyze much
        if (!candidate) {
            return res.status(400).json({ error: 'Please build or upload an ATS Resume first to analyze skill gaps.' });
        }

        let feedbackContext = '';
        if (interviews.length > 0) {
            const evals = interviews.filter(i => i.evaluation).map(i => i.evaluation.feedback).join(' | ');
            feedbackContext = `Interview Feedback: ${evals}`;
        }

        if (!process.env.GEMINI_API_KEY) {
            return res.json({
                skillGaps: ["Advanced React Patterns", "Docker Deployment"],
                recommendations: [
                    { title: "React Advanced Crash Course", reason: "Improve frontend architecture skills." },
                    { title: "Docker & Kubernetes Basics", reason: "Essential for DevOps knowledge missing in your ATS." }
                ]
            });
        }

        const model = genAI.getGenerativeModel({ model: "gemini-1.5-flash" });
        const prompt = `
You are an expert AI Career Coach. 
Analyze this student's profile and identify 2-3 specific technical "Skill Gaps" that are preventing them from getting a Senior/Mid level role.
Then, recommend 2-3 fictional micro-course titles they should take to fix these gaps.

=== STUDENT PROFILE ===
Skills: ${candidate.skills.join(', ')}
Category: ${candidate.category}
ATS Score: ${candidate.atsScore}/100
${feedbackContext}

=== OUTPUT FORMAT ===
Return ONLY JSON without markdown formatting.
{
  "skillGaps": ["Gap 1", "Gap 2"],
  "recommendations": [
    { "title": "Micro-Course Title", "reason": "Why this fixes the gap" }
  ]
}
`;

        const result = await model.generateContent(prompt);
        let cleanJson = result.response.text().replace(/\`\`\`json/gi, '').replace(/\`\`\`/g, '').trim();
        const data = JSON.parse(cleanJson);

        res.json(data);
    } catch (error) {
        console.error("Skill Gap Analysis Error:", error);
        res.status(500).json({ error: 'Failed to analyze skill gaps.' });
    }
});

/**
 * @route   GET /api/candidates/admin/placement-readiness
 * @desc    Predictive Placement Analytics for Admins
 * @access  Private (Admin/Staff)
 */
router.get('/admin/placement-readiness', authenticate, authorize('ADMIN', 'SUPER_ADMIN', 'STAFF'), async (req, res, next) => {
    try {
        // Fetch candidates with ATS score and interview scores
        const candidates = await prisma.candidateProfile.findMany({
            where: { status: { not: 'HIRED' } },
            include: {
                user: {
                    include: {
                        interviews: {
                            include: { evaluation: true }
                        }
                    }
                }
            }
        });

        let hotCandidates = 0;
        let warmCandidates = 0;
        let atRiskCandidates = 0;

        const detailedList = [];

        candidates.forEach(c => {
            // Calculate a composite readiness score
            const atsScore = c.atsScore || 50;
            
            let interviewAvg = 50;
            if (c.user && c.user.interviews && c.user.interviews.length > 0) {
                const evals = c.user.interviews.filter(i => i.evaluation && i.evaluation.overallScore);
                if (evals.length > 0) {
                    const sum = evals.reduce((acc, i) => acc + i.evaluation.overallScore, 0);
                    interviewAvg = sum / evals.length;
                }
            }

            // Simple Model: 40% ATS + 60% Interview
            const probabilityScore = (atsScore * 0.4) + (interviewAvg * 0.6);

            if (probabilityScore >= 80) hotCandidates++;
            else if (probabilityScore >= 60) warmCandidates++;
            else atRiskCandidates++;

            // Only push top hot candidates for the list
            if (probabilityScore >= 80) {
                detailedList.push({
                    id: c.id,
                    name: c.name,
                    category: c.category,
                    probabilityScore: Math.round(probabilityScore)
                });
            }
        });

        res.json({
            summary: {
                hot: hotCandidates,
                warm: warmCandidates,
                atRisk: atRiskCandidates,
                total: candidates.length
            },
            topReadyCandidates: detailedList.sort((a, b) => b.probabilityScore - a.probabilityScore).slice(0, 5)
        });
    } catch (error) {
        console.error("Placement Readiness Error:", error);
        res.status(500).json({ error: 'Failed to load predictive analytics.' });
    }
});
