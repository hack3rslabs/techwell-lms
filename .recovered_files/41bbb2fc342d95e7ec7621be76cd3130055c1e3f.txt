services:
  # ── 1. Schema sync (runs once, then exits) ────────────────────────────────
  migrate:
    build: ./backend
    command: sh -c "npx prisma generate && npx prisma db push --skip-generate"
    environment:
      DATABASE_URL: ${DATABASE_URL:?DATABASE_URL is required}
      NODE_ENV: production
    restart: "no"
    # Do not include in healthcheck-based service meshes
    labels:
      com.docker.compose.exclude_from_hc: "true"

  # ── 2. Backend API ─────────────────────────────────────────────────────────
  backend:
    build: ./backend
    environment:
      DATABASE_URL: ${DATABASE_URL:?DATABASE_URL is required}
      JWT_SECRET: ${JWT_SECRET:?JWT_SECRET is required}
      JWT_EXPIRES_IN: ${JWT_EXPIRES_IN:-7d}
      FRONTEND_URL: ${FRONTEND_URL:?FRONTEND_URL is required}
      NODE_ENV: production
      PORT: 5000
      RAZORPAY_KEY_ID: ${RAZORPAY_KEY_ID:-}
      RAZORPAY_KEY_SECRET: ${RAZORPAY_KEY_SECRET:-}
      RAZORPAY_WEBHOOK_SECRET: ${RAZORPAY_WEBHOOK_SECRET:-}
      GOOGLE_GENERATIVE_AI_KEY: ${GOOGLE_GENERATIVE_AI_KEY:-}
      OPENAI_API_KEY: ${OPENAI_API_KEY:-}
      SMTP_HOST: ${SMTP_HOST:-}
      SMTP_PORT: ${SMTP_PORT:-587}
      SMTP_USER: ${SMTP_USER:-}
      SMTP_PASS: ${SMTP_PASS:-}
      ZOOM_ACCOUNT_ID: ${ZOOM_ACCOUNT_ID:-}
      ZOOM_CLIENT_ID: ${ZOOM_CLIENT_ID:-}
      ZOOM_CLIENT_SECRET: ${ZOOM_CLIENT_SECRET:-}
    depends_on:
      - migrate
    volumes:
      - uploads_prod:/app/uploads
    restart: unless-stopped
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:5000/api/health"]
      interval: 30s
      timeout: 10s
      retries: 5
      start_period: 40s

  # ── 3. Frontend (Next.js) ─────────────────────────────────────────────────
  frontend:
    build: ./frontend
    environment:
      NEXT_PUBLIC_API_URL: ${NEXT_PUBLIC_API_URL:?NEXT_PUBLIC_API_URL is required}
      NODE_ENV: production
    depends_on:
      backend:
        condition: service_healthy
    restart: unless-stopped

volumes:
  uploads_prod:
    driver: local
